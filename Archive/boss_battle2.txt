#!/bin/bash

#  Create Arena_Boss directory
mkdir -p Arena_Boss

# Step 2 & 3: Create 5 text files with random number of lines (10-20)
for i in {1..5}; do
    file="Arena_Boss/file$i.txt"
    lines=$((RANDOM % 11 + 10))  # random number between 10 and 20
    > "$file"  # clear file if exists
    for ((j = 1; j <= lines; j++)); do
        if (( RANDOM % 3 == 0 )); then
            echo "This is a line of Victory!" >> "$file"
        else
            echo "Another line in the arena." >> "$file"
        fi
    done
done
